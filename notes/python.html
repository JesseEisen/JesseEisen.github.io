from contextlib import contextmanager
from pathlib import Path

import os

@contextmanager
def change_directory(path: Path):
    origin = Path().absolute()
    try:
        os.chdir(path)
        yield
    finally:
        os.chdir(origin)

with change_directory('jsme'):
    print(Path().absolute())
